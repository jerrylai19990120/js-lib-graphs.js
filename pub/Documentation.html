APIs Specifications


createHistogram(data, color, title, xAxis, yAxis)

Property Default Type Description
data [] Array An array of numeric values e.g [1, 2, 14, 18]
color dodgerblue String A string of color code or name
title Histogram String Default title for the graph
xAxis x axis String Default x axis name
yAxis y axis String Default y axis name

createPieChart(data, width, height, labels)

Property Default Type Description
data [] Array An array of numeric values e.g [16, 28, 12, 54]
width 400 Number Set a width for the canvas
height 400 Number Set a height for the canvas
title Pie Chart String Default title for the graph
labels [] Array An array of label e.g ["label1", "label2", "label3"]

createAreaGraph(data, width, height, names, title)

Property Default Type Description
data [] Array An nested array of numeric values e.g [[1, 2, 3], [4, 5, 6], [12, 42, 54]]
width null Number Set a width for the canvas
height null Number Set a height for the canvas
names [] Array An array of labels same length as data array e.g ["label1", "label2", "label3"]
title Area Graph String Default title for the graph

createLineChart(data, width, height, color, title, xAixs, yAxis)

Property Default Type Description
data [] Array An array of numeric values e.g [12, 16, 43, 53]
width null Number Set a width for the canvas
height null Number Set a height for the canvas
color red String A string of color code for the line
title Line Chart String Default title for the graph
xAxis x axis String Default x axis name
yAxis y axis String Default y axis name

createRadarGraph(data, width, height, radius, title, dataColors, labels)

Property Default Type Description
data [] Array An nested array of numeric values e.g [[60, 50, 40, 90, 80], [60, 60, 60, 0, 0]]
width null Number Set a width for the canvas
height null Number Set a height for the canvas
radius null Number Set radius for the radar shape
title Radar Graph String Default title for the graph
dataColors [] Array An array of color code string that matches the length of data e.g ["red", "blue"]
labels [] Array An array of string e.g ["label1", "label2"]

createStackedHistogram(data, width, height, title, name)

Property Default Type Description
data [] Array An nested array of numeric values e.g [[1, 2, 4], [5, 2, 3]]
width null Number Set a width for the canvas
height null Number Set a height for the canvas
title Stacked Histogram String Default title for the graph
name [] Array An array of labels has the same length as the nested data arrays e.g ["label1", "label2", "label3"]

createScatterPlot(data, color, title, xAxis, yAxis)

Property Default Type Description
data [] Array An array pf numeric values e.g [1, 3, 43, 41, 54, 66]
color red String A color code string
title Scatter Plot String Default title for the graph
xAxis x axis String Set a name for x axis
yAxis y axis String Set a name for y axis

createPyramidGraph(data, color, title, categories, groups)

Property Default Type Description
data [] Array An nested array of numeric values (nested array with length 2)e.g [[52, 66], [55, 32], [23, 43]]
color dodgerblue String A color code string e.g "blue", "#0000"
title Pyramid Graph String Default title for the graph
categories [] Array An array of string e.g ["height", "weight", "age"]
groups ["1", "2"] Array An array of strings (length 2) e.g ["groupA", "groupB"]
